        // Строка подключения — берём один раз
        string connStr = ConfigurationManager
            .ConnectionStrings["DefaultConnection"].ConnectionString;

        public MainWindow()
        {
            InitializeComponent();
            SetupUI();      // Настройка интерфейса по роли
            LoadProducts(); // Загрузка товаров
        }

        // === НАСТРОЙКА ИНТЕРФЕЙСА ===
        void SetupUI()
        {
            var user = LoginWindow.UserData;
            txtUserInfo.Text = $"{user.Role}: {user.FullName}";

            // Скрываем/показываем элементы по роли
            if (user.Role == "Авторизированный клиент")
            {
                btnOrders.Visibility = Visibility.Collapsed;
                panelFilters.Visibility = Visibility.Collapsed;
                panelAdmin.Visibility = Visibility.Collapsed;
            }
            else if (user.Role == "Менеджер")
            {
                panelFilters.Visibility = Visibility.Visible;
                panelAdmin.Visibility = Visibility.Collapsed;
            }
            else if (user.Role == "Администратор")
            {
                panelFilters.Visibility = Visibility.Visible;
                panelAdmin.Visibility = Visibility.Visible;
            }
            else // Гость
            {
                btnOrders.Visibility = Visibility.Collapsed;
                panelFilters.Visibility = Visibility.Collapsed;
                panelAdmin.Visibility = Visibility.Collapsed;
            }
        }

        // === ЗАГРУЗКА ТОВАРОВ (с фильтрами) ===
        void LoadProducts(string search = "", string category = "", string sort = "")
        {
            var list = new List<object>();

            using (SqlConnection conn = new SqlConnection(connStr))
            {
                string sql = @"SELECT p.Article, p.Name, p.Price, p.Discount, 
                                       p.Quantity, c.Name as Category, 
                                       s.Name as Supplier, m.Name as Manufacturer,
                                       p.Description, p.Id
                                FROM Products p
                                LEFT JOIN Categories c ON p.CategoryId = c.Id
                                LEFT JOIN Suppliers s ON p.SupplierId = s.Id
                                LEFT JOIN Manufacturers m ON p.ManufacturerId = m.Id
                                WHERE 1=1";

                if (!string.IsNullOrEmpty(search))
                    sql += " AND (p.Name LIKE @s OR p.Article LIKE @s)";
                if (category != "" && category != "Все категории")
                    sql += " AND c.Name = @c";

                switch (sort)
                {
                    case "Цена &#8593;": sql += " ORDER BY p.Price ASC"; break;
                    case "Цена &#8595;": sql += " ORDER BY p.Price DESC"; break;
                    case "Название": sql += " ORDER BY p.Name"; break;
                    default: sql += " ORDER BY p.Id"; break;
                }

                SqlCommand cmd = new SqlCommand(sql, conn);
                if (!string.IsNullOrEmpty(search))
                    cmd.Parameters.AddWithValue("@s", "%" + search + "%");
                if (category != "" && category != "Все категории")
                    cmd.Parameters.AddWithValue("@c", category);

                conn.Open();
                SqlDataReader r = cmd.ExecuteReader();
                while (r.Read())
                {
                    list.Add(new
                    {
                        Article = r.GetString(0),
                        Name = r.GetString(1),
                        Price = r.GetDecimal(2),
                        Discount = r.GetInt32(3),
                        Quantity = r.GetInt32(4),
                        Category = r.IsDBNull(5) ? "" : r.GetString(5),
                        Supplier = r.IsDBNull(6) ? "" : r.GetString(6),
                        Manufacturer = r.IsDBNull(7) ? "" : r.GetString(7),
                        Description = r.IsDBNull(8) ? "" : r.GetString(8),
                        Id = r.GetInt32(9)
                    });
                }
            }
            dgProducts.ItemsSource = list;
        }

        // === ФИЛЬТРЫ ===
        void btnApply_Click(object sender, RoutedEventArgs e)
        {
            string cat = (cmbCategory.SelectedItem as ComboBoxItem)?.Content.ToString() ?? "";
            string sort = (cmbSort.SelectedItem as ComboBoxItem)?.Content.ToString() ?? "";
            LoadProducts(txtSearch.Text.Trim(), cat, sort);
        }

     
        // === ДОБАВИТЬ ТОВАР ===
        void btnAdd_Click(object sender, RoutedEventArgs e)
        {
            // Открываем окно с Id = 0 (режим добавления)
            ProductWindow window = new ProductWindow(0);
            window.ShowDialog(); // ShowDialog ждет, пока окно не закроется
            LoadProducts();      // Обновляем таблицу
        }

        // === РЕДАКТИРОВАТЬ ТОВАР ===
        void btnEdit_Click(object sender, RoutedEventArgs e)
        {
            dynamic selected = dgProducts.SelectedItem;
            if (selected == null)
            {
                MessageBox.Show("Сначала выберите товар из списка!");
                return;
            }

            // Открываем окно и передаем Id выбранного товара
            ProductWindow window = new ProductWindow(selected.Id);
            window.ShowDialog();
            LoadProducts(); // Обновляем таблицу после закрытия окна
        }

        // === УДАЛИТЬ ТОВАР ===
        void btnDelete_Click(object sender, RoutedEventArgs e)
        {
            dynamic selected = dgProducts.SelectedItem;
            if (selected == null) { MessageBox.Show("Выберите товар!"); return; }

            if (MessageBox.Show($"Удалить {selected.Name}?", "Подтверждение",
                MessageBoxButton.YesNo) == MessageBoxResult.Yes)
            {
                using (SqlConnection conn = new SqlConnection(connStr))
                {
                    SqlCommand cmd = new SqlCommand("DELETE FROM Products WHERE Id = @id", conn);
                    cmd.Parameters.AddWithValue("@id", selected.Id);
                    conn.Open();
                    cmd.ExecuteNonQuery();
                }
                LoadProducts();
            }
        }

        // === ЗАКАЗЫ ===
        void btnOrders_Click(object sender, RoutedEventArgs e)
        {
            // Заглушка — реализовать по аналогии
            MessageBox.Show("Окно заказов");
        }

        // === ВЫХОД ===
        void btnExit_Click(object sender, RoutedEventArgs e)
        {
            LoginWindow.UserData = (0, "", "", "");
            new LoginWindow().Show();
            this.Close();
        }
    }

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows;
using System.Windows.Controls;
using System.Windows.Data;
using System.Windows.Documents;
using System.Windows.Input;
using System.Windows.Media;
using System.Windows.Media.Imaging;
using System.Windows.Shapes;
using System.Data.SqlClient;
using System.Configuration;

using System.Data;

namespace ShopApp
{
    public partial class ProductWindow : Window
    {
        int currentId = 0;
        string connStr = ConfigurationManager.ConnectionStrings["DefaultConnection"].ConnectionString;

        // Единый конструктор
        public ProductWindow(int id = 0)
        {
            InitializeComponent();
            currentId = id;

            // Загружаем списки справочников при инициализации
            LoadDictionaries();

            if (currentId > 0)
            {
                Title = "Редактирование товара";
                LoadProductData();
            }
            else
            {
                Title = "Добавление товара";
            }
        }

        private void LoadDictionaries()
        {
            FillCombo(cmbCategory, "SELECT Id, Name FROM Categories");
            FillCombo(cmbSupplier, "SELECT Id, Name FROM Suppliers");
            FillCombo(cmbManufacturer, "SELECT Id, Name FROM Manufacturers");
        }

        private void FillCombo(ComboBox combo, string sql)
        {
            using (SqlConnection conn = new SqlConnection(connStr))
            {
                var dt = new DataTable();
                conn.Open();
                new SqlDataAdapter(sql, conn).Fill(dt);
                combo.ItemsSource = dt.DefaultView;
            }
        }

        private void LoadProductData()
        {
            using (SqlConnection conn = new SqlConnection(connStr))
            {
                string sql = "SELECT Name, Price, Discount, Quantity, Description, CategoryId, SupplierId, ManufacturerId FROM Products WHERE Id = @id";
                SqlCommand cmd = new SqlCommand(sql, conn);
                cmd.Parameters.AddWithValue("@id", currentId);

                conn.Open();
                using (SqlDataReader r = cmd.ExecuteReader())
                {
                    if (r.Read())
                    {
                        txtName.Text = r.GetString(0);
                        txtPrice.Text = r.GetDecimal(1).ToString();
                        txtDiscount.Text = r.GetInt32(2).ToString();
                        txtQuantity.Text = r.GetInt32(3).ToString();
                        txtDescription.Text = r.IsDBNull(4) ? "" : r.GetString(4);

                        // Установка значений для ComboBox
                        cmbCategory.SelectedValue = r.IsDBNull(5) ? null : (object)r.GetInt32(5);
                        cmbSupplier.SelectedValue = r.IsDBNull(6) ? null : (object)r.GetInt32(6);
                        cmbManufacturer.SelectedValue = r.IsDBNull(7) ? null : (object)r.GetInt32(7);
                    }
                }
            }
        }

        private void btnSave_Click(object sender, RoutedEventArgs e)
        {
            if (string.IsNullOrWhiteSpace(txtName.Text) || !decimal.TryParse(txtPrice.Text, out decimal price))
            {
                MessageBox.Show("Проверьте поля Название и Цена!");
                return;
            }

            int.TryParse(txtDiscount.Text, out int discount);
            int.TryParse(txtQuantity.Text, out int quantity);

            using (SqlConnection conn = new SqlConnection(connStr))
            {
                string sql = (currentId == 0)
                    ? @"INSERT INTO Products (Article, Name, Price, Discount, Quantity, Description, CategoryId, SupplierId, ManufacturerId) 
                        VALUES (@a, @n, @p, @d, @q, @desc, @cat, @sup, @man)"
                    : @"UPDATE Products SET Name = @n, Price = @p, Discount = @d, Quantity = @q, Description = @desc, 
                        CategoryId = @cat, SupplierId = @sup, ManufacturerId = @man WHERE Id = @id";

                SqlCommand cmd = new SqlCommand(sql, conn);

                // Исправлено: генерация уникального артикула через Guid
                if (currentId == 0)
                {
                    string uniqueArticle = "NEW" + Guid.NewGuid().ToString().Substring(0, 8).ToUpper();
                    cmd.Parameters.AddWithValue("@a", uniqueArticle);
                }

                cmd.Parameters.AddWithValue("@n", txtName.Text);
                cmd.Parameters.AddWithValue("@p", price);
                cmd.Parameters.AddWithValue("@d", discount);
                cmd.Parameters.AddWithValue("@q", quantity);
                cmd.Parameters.AddWithValue("@desc", txtDescription.Text);

                // Используем DBNull, если в ComboBox ничего не выбрано
                cmd.Parameters.AddWithValue("@cat", cmbCategory.SelectedValue ?? (object)DBNull.Value);
                cmd.Parameters.AddWithValue("@sup", cmbSupplier.SelectedValue ?? (object)DBNull.Value);
                cmd.Parameters.AddWithValue("@man", cmbManufacturer.SelectedValue ?? (object)DBNull.Value);

                if (currentId > 0) cmd.Parameters.AddWithValue("@id", currentId);

                try
                {
                    conn.Open();
                    cmd.ExecuteNonQuery();
                    this.Close();
                }
                catch (Exception ex)
                {
                    MessageBox.Show("Ошибка при сохранении: " + ex.Message);
                }
            }
        }
    }
